#A python program to calculate distance between two points.
#Developed By: ARJUN N S

import math
x1, y1 = 4, 2
x2, y2 = 10, 6
distance = math.sqrt((x2 - x1)**2 + (y2 - y1)**2)
print("{:.2f}".format(distance))
